118  std::string read(bool wait = false) {
119  std::string msg("");
120  std::unique_lock<std::mutex> ul(msg_lock_);
121  if (!cache_.empty()) {
122  msg = std::move(cache_.front());
123  cache_.pop_front();
124  }
125 
126  if (!wait) {
127  return msg;
128  }
129 
130  msg_cond_.wait(ul, [this] { return !this->cache_.empty(); });
131  if (!cache_.empty()) {
132  msg = std::move(cache_.front());
133  cache_.pop_front();
134  }
135 
136  return msg;
137  }
138 
139  private:
140  void cb(const std::shared_ptr<const message::PyMessageWrap>& message) {
141  {
142  std::lock_guard<std::mutex> lg(msg_lock_);
143  cache_.push_back(message->data());
144  }
145  if (func_) {
146  func_(channel_name_.c_str());
147  }
148  msg_cond_.notify_one();
149  }
